advanced usage ... 


import saucepan as sauce

def handler_generator(ctx):
  
  def handle1(ctx):
    return "Handler 1"

  def handle2(ctx):
    return "Handler 2"

  if "test" in ctx.request.path:
    return handle1
  elif "xxx" in ctx.request.path:
    return handle2
  else:
    return None
  
sauce.pan.add_router(handler_generator, route_type=sauce.ROUTE_GENERATOR)

@sauce.pan.route(None):
def default_handler(ctx):
  ctx.response.status_code = 403
  return "Nope !"

@sauce.pan.handle_exception(Exception)
def error_handler(ctx, err):
  return "Server fail : " + str(err)

sauce.run(port=8080)
